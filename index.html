<script>

    const GET_URL = "PASTE_GET_URL"; // URL Web App untuk ambil premi

    const POST_URL = "PASTE_POST_URL"; // URL Web App untuk simpan data

    let premiData = [];


    // Ambil data premi dari Google Sheet

    fetch(GET_URL)

        .then(response => response.json())

        .then(data => premiData = data)

        .catch(error => console.error("Error ambil premi:", error));


    function getPremi(gender, usia, plan) {

        for (let row of premiData) {

            if (row.Gender === gender && row.Usia.includes(usia)) {

                return row[plan] || 0;

            }

        }

        return 0;

    }


    document.getElementById("plan").addEventListener("change", updatePremi);

    document.getElementById("usia").addEventListener("input", updatePremi);

    document.getElementById("gender").addEventListener("change", updatePremi);


    function updatePremi() {

        const gender = document.getElementById("gender").value;

        const usia = parseInt(document.getElementById("usia").value);

        const plan = document.getElementById("plan").value;

        if (!gender || !usia || !plan) return;


        const premi = getPremi(gender, usia, plan);

        document.getElementById("premi").value = premi ? premi.toLocaleString("id-ID") : "Data tidak tersedia";

    }


    document.getElementById("insuranceForm").addEventListener("submit", e => {

        e.preventDefault();

        const formData = new FormData(e.target);

        const data = Object.fromEntries(formData.entries());


        fetch(POST_URL, {

            method: "POST",

            body: JSON.stringify(data)

        })

        .then(response => response.text())

        .then(result => alert(result))

        .catch(error => console.error("Error:", error));

    });

</script>
